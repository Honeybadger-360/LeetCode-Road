#最简单的思路
```shell
class Solution:
    def reverse(self, x: int) -> int:
        yus = 0
        fs = False
        if x < 0:
            fs = True
            x = 0 - x
        while x / 10 != 0:
            yus = yus * 10 + x % 10
            x //= 10
        if fs:
            yus = 0 - yus
        return yus if -2147483648 < yus < 2147483647 else 0
```

```shell
class Solution:
    def reverse(self, x: int) -> int:
        n = str(x)
        flag = 0
        if n[0] == "+":
            n=n[1:]
        elif n[0] == "-":
            n=n[1:]
            flag = 1
        n = n[::-1]
        fzs = 0
        for i in n:
            fzs *= 10
            fzs += int(i)
        if flag:
            fzs *= -1
        if (fzs < -pow(2,31)) | (fzs > pow(2, 31)):    
            return 0

        return fzs
```				

```shell
class Solution:
    def reverse(self, x: int) -> int:
        if x > 0:
            n = int(str(x)[::-1])
        else:
            n = - int(str(-x)[::-1])
        if (n < -pow(2,31)) | (n > pow(2, 31)):
```
